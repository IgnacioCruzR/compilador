// --- Prueba Final: Juego de Adivinar el Número ---
// Este programa combina todas las características del lenguaje:
// variables, funciones, E/S, bucles y condicionales.

funcion entero jugar() {
    entero numero_secreto;
    numero_secreto = 42; // El número que el usuario debe adivinar

    entero intento;
    intento = 0; // El número que ingresa el usuario, se inicia en 0 para que el bucle comience

    entero contador_intentos;
    contador_intentos = 0;
    
    imprimir("¡He pensado en un número entre 1 y 100!");
    imprimir("Intenta adivinar cuál es.");

    // El juego continúa mientras el intento del usuario sea diferente al número secreto
    while (intento != numero_secreto) {
        imprimir("Ingresa tu número:");
        
        intento = leer(); // El programa se detendrá aquí para esperar al usuario
        
        contador_intentos = contador_intentos + 1;

        if (intento < numero_secreto) {
            imprimir("¡Muy bajo! Intenta de nuevo.");
        } else {
            if (intento > numero_secreto) {
                imprimir("¡Muy alto! Intenta de nuevo.");
            }
        }
    }

    // El bucle termina cuando el usuario adivina el número
    imprimir("¡CORRECTO! Adivinaste el número.");
    imprimir("Lo lograste en:");
    imprimir(contador_intentos);
    imprimir("intentos.");
    
    retornar 1; // Retornar 1 para indicar que el juego terminó con éxito
}

// --- Código Principal ---
imprimir("--- Iniciando Juego Final de Prueba ---");

jugar(); // Se llama a la función principal del juego

imprimir("--- Juego Terminado. El compilador funciona. ---");
