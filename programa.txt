// --- Prueba de Recursión: Cálculo de Factorial ---

// Definición de la función factorial
// n! = n * (n-1)!
// Caso base: 0! = 1 y 1! = 1
funcion entero factorial(entero n) {
    // Caso base: si n es menor que 2, el factorial es 1.
    if (n < 2) {
        retornar 1;
    } 
    // Caso recursivo: n * factorial(n-1)
    else {
        retornar n * factorial(n - 1);
    }
}


// --- Código Principal ---
imprimir("--- Probando la función factorial recursiva ---");

// Probemos con un número estándar: 5! = 120
entero num;
num = 5;
entero resultado_fact;
resultado_fact = factorial(num);

imprimir("Calculando el factorial de 5...");
imprimir(resultado_fact);


// Probemos con un número un poco más grande: 7! = 5040
imprimir("Calculando el factorial de 7...");
resultado_fact = factorial(7);
imprimir(resultado_fact);


// Probemos directamente el caso base: 1! = 1
imprimir("Calculando el factorial de 1 (caso base)...");
resultado_fact = factorial(1);
imprimir(resultado_fact);
