// --- Calculadora en Mi Lenguaje ---
// Demostración para el taller.

// Variables globales que usaremos como "parámetros" y "valores de retorno"
entero op1;
entero op2;
entero resultado;
cadena operacion;

// --- "Función" para Sumar ---
// Simula: funcion_sumar(100, 50)
imprimir("--- Probando funcion_sumar ---");
op1 = 100;
op2 = 50;
operacion = "Suma";
resultado = op1 + op2;
imprimir("Operacion:");
imprimir(operacion);
imprimir("Resultado:");
imprimir(resultado); // Debe ser 150
imprimir("");

// --- "Función" para Restar ---
// Simula: funcion_restar(100, 30)
imprimir("--- Probando funcion_restar ---");
op1 = 100;
op2 = 30;
operacion = "Resta";
resultado = op1 - op2;
imprimir("Operacion:");
imprimir(operacion);
imprimir("Resultado:");
imprimir(resultado); // Debe ser 70
imprimir("");

// --- "Función" para Exponente usando WHILE ---
// Simula: funcion_exponente(2, 5) => 2^5
imprimir("--- Probando funcion_exponente con bucle WHILE ---");
op1 = 2; // Base
op2 = 5; // Exponente
operacion = "Exponente (2^5)";

// Lógica del bucle
resultado = 1;      // El resultado empieza en 1
entero contador;
contador = op2;     // Usamos un contador para el bucle

while (contador > 0) {
    resultado = resultado * op1; // resultado = resultado * base
    contador = contador - 1;     // Decrementamos el contador
}

imprimir("Operacion:");
imprimir(operacion);
imprimir("Resultado:");
imprimir(resultado); // Debe ser 32
imprimir("");

imprimir("--- Calculadora finalizada con éxito ---");
